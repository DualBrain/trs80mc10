10 REM MC-10 PAINT
20 REM (C) 2013 BY RICHARD VERMEULEN
40 CLS 0
50 PRINT@0,"MC-10 PAINT (C) 2013 RICHARD V"
60 X=1:Y=1:C=1
100 A$=INKEY$
110 IF A$="W" AND Y>0 THEN Y=Y-1
120 IF A$="Z" AND Y<32 THEN Y=Y+1
130 IF A$="A" AND X>0 THEN X=X-1
140 IF A$="S" AND X<63 THEN X=X+1
150 IF A$=" " THEN C=C+1
155 IF A$=CHR$(13) THEN C=9
160 IF C>9 THEN C=0
170 IF C=9 THEN RESET (X,Y)
180 IF C<9 THEN SET(X,Y,C)
190 GOTO 100
